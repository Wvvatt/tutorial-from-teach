保存并播放照片

第 12 课 · 预计 50 分钟 · 最终项目可见成果

本课成果：Mac 接收 JPEG 后先写临时文件，再原子发布为按 UTC 时间排序的 .jpg；浏览器读取最近 60 张照片，并按时间顺序自动播放。
先降低产生速度：你当前的 kUploadIntervalMs 是 1000。先改成 5000 或 10000，否则每分钟会产生约 60 个文件。

1. 数据流从“验证”延长到“发布”

ESP32 POST /images
    ↓ 验证长度与 JPEG 标记
写入 .tmp 临时文件
    ↓ 文件关闭并完整
os.replace() 原子发布为 .jpg
    ↓
GET /api/images 返回时间顺序列表
    ↓
浏览器 GET /images/文件名.jpg 并播放

只让 .jpg 出现在公开列表中。播放器永远看不到正在写入的临时文件。

<body><main>

  <img id="photo" alt="尚无照片">
  <div class="bar">
    <button id="prev">上一张</button>

    <span id="meta">等待照片…</span>

<script>
  let images = [];
  let index = 0;
  let playing = true;

  const photo = document.querySelector('#photo');
  const meta = document.querySelector('#meta');

  function render() {
    if (images.length === 0) {
      photo.removeAttribute('src');
      meta.textContent = '等待照片…';
      return;
    }
    const image = images[index];
    photo.src = image.url;
    meta.textContent = `${index + 1} / ${images.length} · ${image.name}`;
  }

  function move(step) {
    if (images.length === 0) return;
    index = (index + step + images.length) % images.length;
    render();
  }

  async function refreshList() {
    const currentName = images[index]?.name;
    const response = await fetch('/api/images', {cache: 'no-store'});
    const result = await response.json();
    images = result.images;

    const previousIndex = images.findIndex(image => image.name === currentName);
    index = previousIndex >= 0 ? previousIndex : 0;
    render();
  }

  document.querySelector('#prev').onclick = () => move(-1);
  document.querySelector('#next').onclick = () => move(1);
  document.querySelector('#toggle').onclick = event => {
    playing = !playing;
    event.target.textContent = playing ? '暂停' : '播放';
  };

  setInterval(() => { if (playing) move(1); }, 1000);
  setInterval(refreshList, 5000);
  refreshList();
</script>
</body></html>

3. 把接收器升级为图片存储与查询服务

把 server/server.py 替换为下面的完整版本：

import json
import os
import uuid
from datetime import datetime, timezone
from http.server import BaseHTTPRequestHandler, ThreadingHTTPServer
from pathlib import Path
from urllib.parse import unquote, urlparse


ROOT = Path(__file__).resolve().parent
IMAGE_DIR = ROOT / "images"
INDEX_FILE = ROOT / "index.html"
MAX_IMAGE_BYTES = 500_000
MAX_LISTED_IMAGES = 60
IMAGE_DIR.mkdir(exist_ok=True)


class Handler(BaseHTTPRequestHandler):
    def send_bytes(self, status, content_type, body, cache_control="no-store"):
        self.send_response(status)
        self.send_header("Content-Type", content_type)
        self.send_header("Content-Length", str(len(body)))
        self.send_header("Cache-Control", cache_control)
        self.end_headers()
        self.wfile.write(body)

    def send_json(self, status, payload):
        body = (json.dumps(payload) + "\n").encode("utf-8")
        self.send_bytes(status, "application/json", body)

    def do_POST(self):
        if self.path == "/events":
            length = int(self.headers.get("Content-Length", "0"))
            body = self.rfile.read(length)
            print("event:", body.decode("utf-8"))
            self.send_json(201, {"accepted": True})
            return

        if self.path != "/images":
            self.send_error(404)
            return

        content_type = self.headers.get_content_type()
        length = int(self.headers.get("Content-Length", "0"))
        sequence = self.headers.get("X-Image-Sequence", "?")

        if content_type != "image/jpeg":
            self.send_json(415, {"error": "expected image/jpeg"})
            return
        if length < 4 or length > MAX_IMAGE_BYTES:
            self.send_json(413, {"error": "invalid image size"})
            return

        image = self.rfile.read(length)
        complete = len(image) == length
        jpeg = image.startswith(b"\xff\xd8") and image.endswith(b"\xff\xd9")
        if not complete or not jpeg:
            self.send_json(422, {"error": "invalid jpeg"})
            return

        received_at = datetime.now(timezone.utc)
        timestamp = received_at.strftime("%Y%m%dT%H%M%S.%fZ")
        filename = f"{timestamp}-{uuid.uuid4().hex[:6]}.jpg"
        final_path = IMAGE_DIR / filename
        temporary_path = IMAGE_DIR / f".{filename}.tmp"

        try:
            with temporary_path.open("xb") as output:
                output.write(image)
                output.flush()
                os.fsync(output.fileno())
            os.replace(temporary_path, final_path)
        except OSError as error:
            temporary_path.unlink(missing_ok=True)
            print(f"image: save failed: {error}")
            self.send_json(500, {"error": "save failed"})
            return

        print(f"image: seq={sequence} file={filename} bytes={len(image)}")
        self.send_json(201, {
            "accepted": True,
            "sequence": sequence,
            "bytes": len(image),
            "received_at": received_at.isoformat(),
            "url": f"/images/{filename}",
        })

    def do_GET(self):
        path = unquote(urlparse(self.path).path)

        if path == "/":
            if not INDEX_FILE.is_file():
                self.send_error(500, "index.html missing")
                return
            self.send_bytes(200, "text/html; charset=utf-8", INDEX_FILE.read_bytes())
            return

        if path == "/api/images":
            files = sorted(IMAGE_DIR.glob("*.jpg"))[-MAX_LISTED_IMAGES:]
            images = [
                {"name": file.name, "url": f"/images/{file.name}"}
                for file in files
            ]
            self.send_json(200, {"images": images})
            return

        if path.startswith("/images/"):
            filename = path.removeprefix("/images/")
            safe_name = Path(filename).name == filename and filename.endswith(".jpg")
            image_path = IMAGE_DIR / filename
            if not safe_name or not image_path.is_file():
                self.send_error(404)
                return
            self.send_bytes(
                200,
                "image/jpeg",
                image_path.read_bytes(),
                "public, max-age=31536000, immutable",
            )
            return

        self.send_error(404)


server = ThreadingHTTPServer(("0.0.0.0", 8000), Handler)
try:
    print(f"server: http://127.0.0.1:8000 images={IMAGE_DIR}")
    server.serve_forever()
except KeyboardInterrupt:
    print("\nserver: stopped")
finally:
    server.server_close()

4. 为什么要临时文件 + 原子发布

错误方式：直接写 2026....jpg
播放器可能在写到一半时打开它 → 截断 JPEG

本课方式：
.2026....jpg.tmp  ← 只有写入线程知道
       ↓ 写完、flush、fsync、close
os.replace(...tmp, ...jpg)
       ↓ 原子切换名字
2026....jpg       ← 此刻才对查询 API 可见

os.replace() 在同一文件系统内提供原子名称替换：并发读取者看到的是“不存在”或“完整文件”，不会看到中间名字对应的半成品。这里的 UTC 时间是服务器接收时间，不是传感器曝光时间。

第三条命令在请求完成后应无输出。页面应按文件名中的 UTC 时间从旧到新播放，且“上一张 / 暂停 / 下一张”可用。

6. 本课刻意留下的限制

  • 磁盘文件会无限增长，还没有保留上限。
  • 服务器重启后仍能从目录恢复列表，但没有数据库或额外元数据。
  • ESP32 上传失败时没有退避、缓存和失败计数。
  • 服务只适合可信局域网，没有认证。

这些限制会成为第 13 课“可靠性与可诊断性”的输入，不要在本课同时解决。

7. 回忆检查

为什么不能直接把上传正文写进最终 .jpg 文件？请在答案中使用“半张图片”“原子”“并发读取”。

通过标准

  • 至少 5 张 JPEG 以 UTC 时间文件名保存在 server/images/。
  • /api/images 返回最近照片的时间顺序列表。
  • 浏览器能自动播放并手动切换照片。
  • 上传过程中反复刷新页面，不出现截断图片或遗留 .tmp。
